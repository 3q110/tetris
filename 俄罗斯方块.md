\# Hermes 俄罗斯方块小游戏

基于Hermes引擎开发的经典俄罗斯方块，包含预览、难度升级、暂停重启完整功能



\## 一、游戏介绍

\- 经典7种方块造型，彩色方块视觉效果

\- 支持方块移动、旋转、加速下落、满行消除

\- 附带下一方块预览、等级难度递增、计分统计

\- 具备开始、暂停、游戏结束、重新开局机制



\## 二、操作说明

| 按键 | 功能 |

| ---- | ---- |

| 空格键 | 开始游戏 / 暂停游戏 / 重新开局 |

| ↑ 方向键 | 旋转方块 |

| ← 方向键 | 方块左移 |

| → 方向键 | 方块右移 |

| ↓ 方向键 | 加速下落 |



\## 三、核心功能清单

✅ 七种经典俄罗斯方块样式

✅ 方块碰撞边界检测

✅ 自动下落与手动操控

✅ 整行消除得分

✅ 得分越高难度自动提速

✅ 右侧下一方块预览

✅ 游戏暂停/继续

✅ 游戏结束重置开局

✅ 实时显示分数与当前等级



\## 四、完整运行代码

```typescript

// Hermes 升级版俄罗斯方块

const canvas = Hermes.canvas;

const ctx = canvas.getContext("2d");



// 基础配置

const ROW = 20;

const COL = 10;

const SIZE = 30;

const PREV\_SIZE = 20;

const GAME\_WIDTH = COL \* SIZE;

const GAME\_HEIGHT = ROW \* SIZE;

const PANEL\_OFFSET = 320;



// 方块形状与配色

const SHAPES = \[

&#x20; \[\[1,1,1,1]],

&#x20; \[\[1,1],\[1,1]],

&#x20; \[\[0,1,0],\[1,1,1]],

&#x20; \[\[1,0,0],\[1,1,1]],

&#x20; \[\[0,0,1],\[1,1,1]],

&#x20; \[\[0,1,1],\[1,1,0]],

&#x20; \[\[1,1,0],\[0,1,1]]

];

const COLORS = \["#0ff","#ff0","#a0f","#f80","#00f","#0f0","#f00"];



// 游戏状态

let board = Array(ROW).fill(0).map(() => Array(COL).fill(0));

let currentShape, currentColor, nextShape, nextColor;

let currentX = 0, currentY = 0;

let score = 0, level = 1, dropSpeed = 800;

let gameOver = false, isPause = false, isStart = false;

let dropTimer;



// 生成随机方块

function createPiece(){

&#x20; const idx = Math.floor(Math.random() \* SHAPES.length);

&#x20; return {shape:SHAPES\[idx], color:COLORS\[idx]};

}



// 初始化新方块

function resetPiece(){

&#x20; const piece = nextShape ? nextShape : createPiece();

&#x20; currentShape = piece.shape;

&#x20; currentColor = piece.color;

&#x20; nextShape = createPiece();

&#x20; nextColor = nextShape.color;

&#x20; currentX = Math.floor(COL / 2) - Math.floor(currentShape\[0].length / 2);

&#x20; currentY = 0;

&#x20; if(collision(currentX,currentY)) gameOver = true;

}



// 碰撞检测

function collision(x,y){

&#x20; for(let r=0;r<currentShape.length;r++){

&#x20;   for(let c=0;c<currentShape\[r].length;c++){

&#x20;     if(currentShape\[r]\[c]){

&#x20;       let nx = x+c, ny = y+r;

&#x20;       if(nx<0||nx>=COL||ny>=ROW) return true;

&#x20;       if(ny>=0 \&\& board\[ny]\[nx]) return true;

&#x20;     }

&#x20;   }

&#x20; }

&#x20; return false;

}



// 方块固化

function merge(){

&#x20; currentShape.forEach((row,r)=>{

&#x20;   row.forEach((val,c)=>{

&#x20;     if(val) board\[currentY+r]\[currentX+c] = currentColor;

&#x20;   })

&#x20; })

}



// 消行+升级难度

function clearLine(){

&#x20; let lineNum = 0;

&#x20; for(let r=ROW-1;r>=0;r--){

&#x20;   if(board\[r].every(v=>v)){

&#x20;     board.splice(r,1);

&#x20;     board.unshift(Array(COL).fill(0));

&#x20;     r++;

&#x20;     lineNum++;

&#x20;   }

&#x20; }

&#x20; if(lineNum){

&#x20;   score += lineNum \* 100 \* level;

&#x20;   let newLv = Math.floor(score / 500) + 1;

&#x20;   if(newLv > level){

&#x20;     level = newLv;

&#x20;     dropSpeed = Math.max(200, 800 - level\*80);

&#x20;     clearInterval(dropTimer);

&#x20;     dropTimer = setInterval(drop, dropSpeed);

&#x20;   }

&#x20; }

}



// 旋转方块

function rotate(){

&#x20; let rotateS = currentShape\[0].map((\_,i)=>currentShape.map(row=>row\[i]).reverse());

&#x20; let old = currentShape;

&#x20; currentShape = rotateS;

&#x20; if(collision(currentX,currentY)) currentShape = old;

}



// 自动下落

function drop(){

&#x20; if(gameOver||isPause||!isStart) return;

&#x20; if(!collision(currentX,currentY+1)) currentY++;

&#x20; else{

&#x20;   merge();

&#x20;   clearLine();

&#x20;   resetPiece();

&#x20; }

}



// 绘制界面

function draw(){

&#x20; ctx.fillStyle="#111";

&#x20; ctx.fillRect(0,0,canvas.width,canvas.height);



&#x20; board.forEach((row,r)=>{

&#x20;   row.forEach((color,c)=>{

&#x20;     if(color){

&#x20;       ctx.fillStyle=color;

&#x20;       ctx.fillRect(c\*SIZE,r\*SIZE,SIZE-1,SIZE-1);

&#x20;     }

&#x20;   })

&#x20; })



&#x20; if(currentShape){

&#x20;   currentShape.forEach((row,r)=>{

&#x20;     row.forEach((val,c)=>{

&#x20;       if(val){

&#x20;         ctx.fillStyle=currentColor;

&#x20;         ctx.fillRect((currentX+c)\*SIZE,(currentY+r)\*SIZE,SIZE-1,SIZE-1);

&#x20;       }

&#x20;     })

&#x20;   })

&#x20; }



&#x20; ctx.fillStyle="#fff";

&#x20; ctx.font="22px Arial";

&#x20; ctx.fillText(`得分：${score}`,PANEL\_OFFSET,40);

&#x20; ctx.fillText(`等级：${level}`,PANEL\_OFFSET,80);

&#x20; ctx.fillText("下一方块",PANEL\_OFFSET,130);



&#x20; if(nextShape){

&#x20;   nextShape.forEach((row,r)=>{

&#x20;     row.forEach((val,c)=>{

&#x20;       if(val){

&#x20;         ctx.fillStyle=nextColor;

&#x20;         ctx.fillRect(PANEL\_OFFSET+c\*PREV\_SIZE,160+r\*PREV\_SIZE,PREV\_SIZE-1,PREV\_SIZE-1);

&#x20;       }

&#x20;     })

&#x20;   })

&#x20; }



&#x20; if(!isStart){

&#x20;   ctx.fillStyle="#fff";

&#x20;   ctx.font="36px Arial";

&#x20;   ctx.fillText("按空格开始",GAME\_WIDTH/2-90,GAME\_HEIGHT/2);

&#x20; }

&#x20; if(isPause\&\&!gameOver){

&#x20;   ctx.fillStyle="#ff0";

&#x20;   ctx.font="36px Arial";

&#x20;   ctx.fillText("游戏暂停",GAME\_WIDTH/2-70,GAME\_HEIGHT/2);

&#x20; }

&#x20; if(gameOver){

&#x20;   ctx.fillStyle="#f33";

&#x20;   ctx.font="40px Arial";

&#x20;   ctx.fillText("游戏结束",GAME\_WIDTH/2-80,GAME\_HEIGHT/2);

&#x20;   ctx.font="24px Arial";

&#x20;   ctx.fillText("空格重新开始",GAME\_WIDTH/2-95,GAME\_HEIGHT/2+50);

&#x20; }

}



// 键盘监听

Hermes.onKeyDown(key=>{

&#x20; if(key==="Space"){

&#x20;   if(gameOver){

&#x20;     resetGame();

&#x20;     return;

&#x20;   }

&#x20;   if(!isStart){

&#x20;     isStart=true;

&#x20;     return;

&#x20;   }

&#x20;   isPause = !isPause;

&#x20;   return;

&#x20; }

&#x20; if(!isStart||isPause||gameOver) return;



&#x20; switch(key){

&#x20;   case "ArrowLeft":!collision(currentX-1,currentY)\&\&currentX--;break;

&#x20;   case "ArrowRight":!collision(currentX+1,currentY)\&\&currentX++;break;

&#x20;   case "ArrowDown":drop();break;

&#x20;   case "ArrowUp":rotate();break;

&#x20; }

})



// 重置游戏

function resetGame(){

&#x20; board = Array(ROW).fill(0).map(()=>Array(COL).fill(0));

&#x20; score=0;level=1;dropSpeed=800;

&#x20; gameOver=false;isPause=false;isStart=false;

&#x20; nextShape=null;

&#x20; resetPiece();

&#x20; clearInterval(dropTimer);

&#x20; dropTimer = setInterval(drop,dropSpeed);

}



// 游戏循环

function gameLoop(){

&#x20; draw();

&#x20; Hermes.nextFrame(gameLoop);

}



// 画布初始化

canvas.width = 480;

canvas.height = GAME\_HEIGHT;

resetPiece();

dropTimer = setInterval(drop,dropSpeed);

gameLoop();
